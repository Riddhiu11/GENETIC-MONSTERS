package org.example;
import java.io.*;
import java.util.ArrayList;

abstract class Monster implements Serializable {
    protected String name;
    protected String color;
    protected int strength;
    protected int speed;

    public Monster(String name, String color, int strength, int speed) {
        this.name = name;
        this.color = color;
        this.strength = strength;
        this.speed = speed;
    }

    public abstract void performSpecialAbility();

    public void saveToFile(String filename) {
        try (BufferedWriter writer = new BufferedWriter(new FileWriter(filename, true))) {
            writer.write(this.toString());
            writer.newLine();
        } catch (IOException e) {
            System.out.println("Error saving monster: " + e.getMessage());
        }
    }

    @Override
    public String toString() {
        return name + "," + color + "," + strength + "," + speed;
    }

    public static class Flyingmonster {
    }

    static class FileManager {
        public static void saveMonsters(ArrayList<Monster> monsters, String filename) {
            try (ObjectOutputStream oos = new ObjectOutputStream(new FileOutputStream(filename))) {
                oos.writeObject(monsters);
                System.out.println("Game saved successfully.");
            } catch (IOException e) {
                System.out.println("Error saving game: " + e.getMessage());
            }
        }

        public static ArrayList<Monster> loadMonsters(String filename) {
            ArrayList<Monster> monsters = new ArrayList<>();
            try (ObjectInputStream ois = new ObjectInputStream(new FileInputStream(filename))) {
                monsters = (ArrayList<Monster>) ois.readObject();
                System.out.println("Game loaded successfully.");
            } catch (IOException | ClassNotFoundException e) {
                System.out.println("Error loading game: " + e.getMessage());
            }
            return monsters;
        }
    }
}
