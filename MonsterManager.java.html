package org.example;
import java.util.ArrayList;

class MonsterManager {
    private ArrayList<Monster> monsters = new ArrayList<>();

    public void addMonster(Monster m) {
        monsters.add(m);
    }

    public void breedMonsters(int index1, int index2) {
        if (index1 < 0 || index2 < 0 || index1 >= monsters.size() || index2 >= monsters.size()) {
            System.out.println("Invalid monster selection.");
            return;
        }
        Monster parent1 = monsters.get(index1);
        Monster parent2 = monsters.get(index2);

        // Simple breeding logic (averaging strengths and speeds)
        String newName = parent1.name + "-" + parent2.name;
        String newColor = "Mixed";
        int newStrength = (parent1.strength + parent2.strength) / 2;
        int newSpeed = (parent1.speed + parent2.speed) / 2;

        Monster offspring = new Monster(newName, newColor, newStrength, newSpeed) {
            @Override
            public void performSpecialAbility() {

            }
        };
        addMonster(offspring);
        System.out.println("New monster created: " + offspring.name);
    }

    public void saveGame(String filename) {
        Monster.FileManager.saveMonsters(monsters, filename);
    }

    public void loadGame(String filename) {
        monsters = Monster.FileManager.loadMonsters(filename);
    }

    public void displayMonsters(boolean detailed) {
        for (Monster m : monsters) {
            if (detailed) {
                System.out.println(m.toString());
            } else {
                System.out.println(m.name);
            }
        }
    }
}
